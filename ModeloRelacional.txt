Usuarios(
    id PK,
    correo AK,
    contraseña,
    nombre
)
Clientes(
    id PK FK → Usuarios.id,
    fechaNacimiento,
    alquilerActivo,
    borrado
)
Tecnicos_Mantenimiento(
    id PK FK → Usuarios.id,
    fechaFinUltimoServicio NULL, -- derivado pero almacenable
    borrado
)
Vehiculos(
    id PK,
    estado, --derivado pero almacenable
    kilometraje,    --derivado pero almacenable
    numeroUsos,    --derivado pero almacenable
    localizacion,  --derivado pero almacenable
    borrado
)
Bicicletas(
    id PK FK → Vehiculos.id,
    tipoBici
)
Patinetes_Electricos(
    id PK FK → Vehiculos.id,
    autonomiaBateria
)
'El nombre de la estación es único porque la app solo se va a usar en Sevilla y no van a coincidir dos nombres de estaciones'
Estaciones(
    id PK,
    nombre AK,
    borrado
)
Enganches(
    id PK,
    estacionId FK → Estaciones.id,
    numero,
    estado,  --derivado pero almacenable
    borrado
)
Alquileres(
    id PK,
    clienteId FK → Clientes.id,
    vehiculoId FK → Vehiculos.id,
    engancheInicioId FK → Enganches.id,
    engancheFinId FK → Enganches.id NULL,
    fechaHoraInicio,
    fechaHoraFin NULL,
    distanciaRecorrida NULL,
    costo NULL,  --derivado pero almacenable
    lugarInicio, --derivado pero almacenable
    lugarFin NULL  --derivado pero almacenable
)
Valoraciones(
    id PK,
    alquilerId FK → Alquileres.id,
    vehiculoId FK → Vehiculos.id,
    puntuacion,
    comentario NULL
)
Reparaciones(
    id PK,
    tecnicoId FK → Tecnicos_Mantenimiento.id,
    vehiculoId FK → Vehiculos.id,
    fecha,
    detalles
)
Pagos(
    id PK,
    clienteId FK → Clientes.id,
    alquilerId FK → Alquileres.id NULL,
    tipoPago,     -- derivado pero almacenable
    cantidad,
    fecha
)
