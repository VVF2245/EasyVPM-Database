    correo AK,
    contraseña
)
CLIENTE(
    clienteId PK FK → USUARIO.usuarioId,
    tarifaActual,
    fechaNacimiento
)
TECNICO_MANTENIMIENTO(
    tecnicoId PK FK → USUARIO.usuarioId,
    fechaUltimoServicio -- derivado pero almacenable
)
VEHICULO(
    vehiculoId PK,
    estado, --derivado pero almacenable
    kilometraje,    --derivado pero almacenable
    numeroUsos,    --derivado pero almacenable
    localizacion  --derivado pero almacenable
)
BICICLETA(
    bicicletaId AK,
    vehiculoId PK FK → VEHICULO.idVehiculo,
    tipoBici
)
PATINETE_ELECTRICO(
    patineteElectricoId
    vehiculoId PK FK → VEHICULO.vehiculoId,
    autonomiaBateria
)

ESTACION(
    estacionId PK,
    nombre AK
)
ENGANCHE(
    engancheId PK,
    estacionId FK → ESTACION.estacionId,
    numero,
    estado  --derivado pero almacenable
)
ALQUILER(
    alquilerId PK,
    clienteId FK → CLIENTE.clienteId,
    vehiculoId FK → VEHICULO.vehiculoId,
    engancheInicioId FK → ENGANCHE.engancheId,
    engancheFinId FK → ENGANCHE.engancheId NULL,
    fechaHoraInicio,
    fechaHoraFin NULL,
    distanciaRecorrida NULL,
    costo NULL,  --derivado pero almacenable
    lugarInicio, --derivado pero almacenable
    lugarFin NULL  --derivado pero almacenable
)
VALORACION(
    valoracionId PK,
    alquilerId FK → ALQUILER.alquilerId,
    vehiculoId FK → VEHICULO.vehiculoId,
    puntuacion,
    comentario NULL
)
REPARACION(
    reparacionId PK,
    tecnicoId FK → TECNICO_MANTENIMIENTO.tecnicoId,
    vehiculoId FK → VEHICULO.vehiculoId,
    fecha,
    detalles
)
PAGO(
    pagoId PK,
    clienteId FK → CLIENTE.clienteId,
    alquilerId FK → ALQUILER.alquilerId NULL,
    tipoPago,     -- derivado pero almacenable
    cantidad,
